爬取网页内容并导出excel

环境

在开始前，需要本地有node环境，可以在终端输入node --version确认。如果没有，请去nodejs官网下载安装。

$ node --version
v15.13.0

代码

安装依赖

npm i superagent -S
npm i superagent-charset -S
npm i cheerio -S
npm i xlsx -S

发起请求

使用superagent来请求网页内容，因为不同页面的编码可能会不同，所以还要用到superagent-charset。

const superagent = require("superagent");
const charset = require("superagent-charset");
charset(superagent); 

/**
 * 发起请求
 * @param {string} url 请求地址
 * @param {string} charset 网站编码方式
 */
function sendRequest(url, charset) {
    superagent
        .get(url)
        .buffer(true)
        .charset(charset)
        .end((_, response) => {
            // response.text 为返回内容
        });
}

处理页面内容

使用cheerio模块来提取网页中的数据。

const cheerio = require("cheerio");

/**
 * 处理网站内容
 * @param {srting} content 网站内容
 */
function formatData(content) {
    const $ = cheerio.load(content);
    // 针对网页内容不同处理
    const dataContainer = Array.from($("#table_lit_bang_105 tr"));
    dataContainer.shift(); // 删除表头

    let excleData = [];
    dataContainer.forEach(value => {
        const td = $(value).find('td');
        excleData.push({
            "序号": td.eq(0).find('span').text(),
            "文章标题": td.eq(1).find('a').html(),
            "文章地址": td.eq(1).find('a').attr('href'),
            "作者": td.eq(2).find('a').html(),
            "时间": td.eq(3).text(),
        });
    });
}

导出excel

const XLSX = require('xlsx');

/**
 * 导出数据
 * @param {srting} excleData 
 */
function exportExcel(excleData) {
    const filename = "./美食每日博文排行.xlsx"; // 存放文件名（含路径）
    const ws_name = "sheet"; // excel工作表名

    // 设置表格样式，!cols为列宽
    const options = {
        '!cols': [
            { wpx: 30 },
            { wpx: 365 },
            { wpx: 320 },
            { wpx: 100 },
            { wpx: 40 }
        ]
    };

    const ws = XLSX.utils.json_to_sheet(excleData);
    const wb = XLSX.utils.book_new();
    ws["!cols"] = options['!cols'];
    XLSX.utils.book_append_sheet(wb, ws, ws_name);
    XLSX.writeFileSync(wb, filename);
}

const fs = require("fs");

function isFileExisted(path) {
    return new Promise((resolve, reject) => {
        fs.access(path, error => {
            if (error) {

                    if (error) {
                        reject(false);

                        console.log(`创建${path}目录成功`);

                    }

            } else {
                resolve(true);
            }
        })
    })
};

function sendRequest(url, charset) {
