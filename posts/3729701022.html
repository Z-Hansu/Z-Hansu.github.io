vue使用i18n

i18n是一个专门用于处理多语言的插件，其来源是英文单词internationalization的首末字符i和n,中间部分nternationalizatio刚好18个字母,故被起名i18n。

安装

使用npm安装:

npm install vue-i18n

在main.js中引用

import VueI18n from 'vue-i18n';

Vue.use(VueI18n);

// 引入语言包，注意路径
import zh from '@/common/languages/zh.js';
import en from '@/common/languages/en.js';

const messages = {
	'en': en,
	'zh': zh
}

const i18n = new VueI18n({
	locale: 'zh', // 默认语言, 值为 messages 的 key 值
	messages,
})

const app = new Vue({
	i18n,
	...App
})

构建语言包

zh.js

// zh.js
export default {
    language: {
		name: '中文'
	},
	common: {
		confirm: '确定',
		next: '下一步',
        // ......
	},
	index: {
		// ......
	}
}

en.js

// en.js
export default {
    language: {
		name: 'English'
	},
	common: {
		confirm: 'confirm',
		next: 'next',
        price: '单价: {price}个'
        // ......
	},
	index: {
		// ......
	}
}

使用

<template>
    <!-- 普通调用 -->
	<view>{{ $t('common.confirm') }}</view>
    <!-- 含有变量 -->
    <view>{{ $t('common.price', { price: goods.price }) }}</view>
</template>

<script>
	export default {
        // js 中使用
		onLoad() {
			console.log(this.$t('common.next'));
		}
	}
</script>

语言切换

<template>
	<view @click="switchLanguage">切换语言</view>
</template>

<script>
	export default {
		methods: {
			switchLanguage() {
				this.$i18n.locale = this.$i18n.locale === 'en' ? 'zh' : 'en';
			}
		}
	}
</script>
